CREATE TABLE CONVIDADO (ID INTEGER NOT NULL, EMAIL VARCHAR(255), EMPRESA VARCHAR(255), NOME VARCHAR(255), VERSAO BIGINT, PRIMARY KEY (ID))
CREATE TABLE PARTICIPANTE (ID INTEGER NOT NULL, EMAIL VARCHAR(255), NOME VARCHAR(255), VERSAO BIGINT, PRIMARY KEY (ID))
CREATE TABLE REUNIAO_PARTICIPANTE (reunioes_ID INTEGER NOT NULL, participantes_ID INTEGER NOT NULL, PRIMARY KEY (reunioes_ID, participantes_ID))
